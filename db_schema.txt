CREATE TABLE ADMIN (
ID INT,
HASH VARCHAR(50));

CREATE TABLE ARCHIVE (
ID INTEGER PRIMARY KEY AUTOINCREMENT,
DEVICE INT,
DIR VARCHAR(50),
FILE VARCHAR(50)
);

CREATE TABLE CERTS (
ID INTEGER PRIMARY KEY AUTOINCREMENT,
DEVICE INT NOT NULL,
NAME VARCHAR(50),
ISSUER VARCHAR(50),
EXPIRE INT NOT NULL,
SN VARCHAR(50),
KEY INT,
SUB_C VARCHAR(4),
SUB_S VARCHAR(30),
SUB_L VARCHAR(30),
SUB_O VARCHAR(50),
SUB_OU VARCHAR(50),
SUB_CN VARCHAR(50) );

CREATE TABLE DEVICES (
ID INTEGER PRIMARY KEY AUTOINCREMENT,
NAME TEXT NOT NULL,
IP CHAR(15),
CID_TIME INT,
DATE_ADDED INT,
LAST_DATE INT,
VERSION VACHAR(10),
BUILD VACHAR(10),
MODEL VACHAR(20),
HOSTNAME VARCHAR(50),
DEV_TYPE VARCHAR(50),
SERIAL VARCHAR(70),
ACT_PARTITION VARCHAR(10),
PASSWD VARCHAR(50)
);

CREATE TABLE JOBS (
ID INTEGER PRIMARY KEY AUTOINCREMENT,
DATE DATE NOT NULL,
TIME INT NOT NULL,
ERRORS INT,
COMPLETE BOOLEAN,
DEVICE_TOTAL INT,
DEVICE_COMPLETE INT,
DEVICE_W_ERRORS TEXT
);

CREATE TABLE MOTD (
ID INT,
MOTD TEXT);

CREATE TABLE SETTINGS_INT (
NAME VARCHAR(30),
VALUE INT);

CREATE TABLE SETTINGS_TEXT (
NAME VARCHAR(30),
VALUE VARCHAR(255)
);

CREATE TABLE USERS (
ID INTEGER PRIMARY KEY AUTOINCREMENT,
NAME VARCHAR(30),
HASH VARCHAR(100),
ADDED_BY VARCHAR(30),
DATE_ADDED INT
);

CREATE TABLE BACKUP_USER (
ID INT,
